# Quick local run (Ollama). Dummy token+key + --bare:
# - Without --bare, interactive TUI often still hits Anthropic OAuth/keychain and shows "Not logged in"
#   even when ANTHROPIC_BASE_URL points at Ollama (`claude --help` documents --bare skipping keychain/OAuth).
# See: https://docs.ollama.com/integrations/claude-code
alias claude-local="ANTHROPIC_AUTH_TOKEN=ollama ANTHROPIC_API_KEY=ollama ANTHROPIC_BASE_URL=${_CLAUDE_OLLAMA_URL} claude --bare --model qwen3.5:4b"

_claude_pick_key() {
  # args: array_name index_var_name
  local array_name="$1"
  local index_var="$2"
  local -a keys
  eval "keys=(\"\${${array_name}[@]}\")"

  if (( ${#keys[@]} == 0 )); then
    echo ""
    return 0
  fi

  # Initialize the rotation index if needed.
  eval ": \${${index_var}:=0}"
  local idx
  eval "idx=\"\${${index_var}}\""
  if (( idx >= ${#keys[@]} )); then
    idx=0
  fi

  # Select key for this call, then advance index for next time.
  local key="${keys[$((idx+1))]}"
  idx=$(( (idx + 1) % ${#keys[@]} ))
  eval "${index_var}=${idx}"
  echo "$key"
}

_nvidia_key_for_model() {
  # Choose a key that can actually serve the selected NVIDIA model.
  # Some keys intermittently timeout or lack access for specific models.
  local model="$1"
  local endpoint="https://integrate.api.nvidia.com/v1/chat/completions"
  local key code idx=1
  for key in "${_NVIDIA_KEYS[@]}"; do
    code="$(curl -m 12 -sS -o /dev/null -w "%{http_code}" \
      "$endpoint" \
      -H "Authorization: Bearer $key" \
      -H "Content-Type: application/json" \
      -d "{\"model\":\"$model\",\"messages\":[{\"role\":\"user\",\"content\":\"ping\"}],\"max_tokens\":8}" 2>/dev/null)"
    if [[ "$code" == "200" ]]; then
      # Keep rotation index aligned with last known-good key.
      _NVIDIA_KEY_INDEX=$((idx % ${#_NVIDIA_KEYS[@]}))
      echo "$key"
      return 0
    fi
    idx=$((idx+1))
  done
  echo ""
  return 1
}

claude-free() {
  if ! command -v claude >/dev/null 2>&1; then
    echo "claude-free: 'claude' command not found."
    return 1
  fi

  local -a options
  local -A providerByChoice
  local -A modelByChoice
  local -i i=1
  local m

  for m in "${_NVIDIA_MODELS[@]}"; do
    options+=("NVIDIA    | ${m}")
    providerByChoice[$i]="nvidia"
    modelByChoice[$i]="$m"
    i=$((i+1))
  done

  for m in "${_OPENROUTER_MODELS[@]}"; do
    options+=("OpenRouter| ${m}")
    providerByChoice[$i]="openrouter"
    modelByChoice[$i]="$m"
    i=$((i+1))
  done

  for m in "${_OLLAMA_MODELS[@]}"; do
    options+=("Ollama    | ${m}")
    providerByChoice[$i]="ollama"
    modelByChoice[$i]="$m"
    i=$((i+1))
  done

  if (( ${#options[@]} == 0 )); then
    echo "claude-free: no models configured."
    return 1
  fi

  echo "Select a model:"
  select opt in "${options[@]}"; do
    [[ -z "$opt" ]] && echo "Invalid selection." && continue
    local choice="$REPLY"
    local provider="${providerByChoice[$choice]}"
    local model="${modelByChoice[$choice]}"
    [[ -z "$provider" || -z "$model" ]] && echo "Invalid selection." && continue

    local base_url api_key
    if [[ "$provider" == "nvidia" ]]; then
      echo "claude-free -> nvidia | model=$model (via claudish proxy)"
      claudish-nvidia "$model" "$@"
      return $?
    elif [[ "$provider" == "openrouter" ]]; then
      base_url="https://openrouter.ai/api/v1"
      api_key="$(_claude_pick_key _OPENROUTER_KEYS _OPENROUTER_KEY_INDEX)"
      [[ -z "$api_key" ]] && echo "claude-free: no OpenRouter keys configured in ~/.claude-keys.zsh" && return 1
    else
      base_url="${_CLAUDE_OLLAMA_URL:-http://localhost:11434}"
      api_key=""
    fi

    echo "claude-free -> $provider | model=$model"

    # Ollama: Claude Code expects *some* Anthropic-compat credentials; use Ollama’s dummy token+key.
    # Unsetting both causes "Not logged in · Please run /login" (especially after `claude /logout`).
    if [[ "$provider" == "ollama" ]]; then
      ANTHROPIC_AUTH_TOKEN=ollama \
      ANTHROPIC_API_KEY=ollama \
      ANTHROPIC_BASE_URL="$base_url" \
        claude --bare --model "$model" "$@"
    else
      env -u ANTHROPIC_AUTH_TOKEN \
        ANTHROPIC_API_KEY="$api_key" \
        ANTHROPIC_BASE_URL="$base_url" \
        claude --bare --model "$model" "$@"
    fi

    return $?
  done
}

# ─── Claudish + NVIDIA NIM (OpenAI-compatible API) ─────────────────────
# Raw `claude` + ANTHROPIC_BASE_URL → integrate.api.nvidia.com often fails because
# Claude Code speaks Anthropic protocol; NVIDIA integrate speaks OpenAI chat/completions.
# Claudish translates Anthropic ↔ OpenAI. Use provider shortcut: oai@<model_id>
# Requires: `npm install -g claudish` and Bun (~/.bun/bin). See: https://github.com/MadAppGang/claudish
claudish-nvidia() {
  export PATH="${BUN_INSTALL:-$HOME/.bun}/bin:$PATH"
  if ! command -v claudish >/dev/null 2>&1; then
    echo "claudish-nvidia: install with: npm install -g claudish  (and Bun: https://bun.sh)"
    return 1
  fi
  local model="${1:-deepseek-ai/deepseek-v3.2}"
  shift
  local api_key
  api_key="$(_nvidia_key_for_model "$model")"
  [[ -z "$api_key" ]] && api_key="$(_claude_pick_key _NVIDIA_KEYS _NVIDIA_KEY_INDEX)"
  if [[ -z "$api_key" ]]; then
    echo "claudish-nvidia: add nvapi keys to ~/.claude-keys.zsh (_NVIDIA_KEYS)"
    return 1
  fi
  echo "claudish-nvidia -> oai@${model} (OPENAI_BASE_URL=NVIDIA integrate)"
  # Placeholder suppresses Claude Code login prompt; real chat goes to NVIDIA via OpenAI client.
  # CLAUDISH_DIAG_MODE=off: skip Claudish's mtm wrapper so the TUI matches plain `claude` (readable).
  # Restore Claudish diagnostics: `CLAUDISH_DIAG_MODE=auto claudish-nvidia` or `--diag-mode auto`.
  env OPENAI_BASE_URL="https://integrate.api.nvidia.com" \
    OPENAI_API_KEY="$api_key" \
    ANTHROPIC_API_KEY="sk-ant-api03-placeholder" \
    CLAUDISH_DIAG_MODE="${CLAUDISH_DIAG_MODE:-off}" \
    claudish --model "oai@${model}" "$@"
}

  if ! command -v claudish >/dev/null 2>&1; then

    return 1
  fi
  local -a options
  local -A modelByChoice
  local -i i=1
  local m
  for m in "${_NVIDIA_MODELS[@]}"; do
    options+=("${m}")
    modelByChoice[$i]="$m"
    i=$((i+1))
  done
  echo "NVIDIA models via Claudish (pick one):"
  select opt in "${options[@]}"; do
    [[ -z "$opt" ]] && echo "Invalid selection." && continue
    local choice="$REPLY"
    local model="${modelByChoice[$choice]}"
    [[ -z "$model" ]] && echo "Invalid selection." && continue
    claudish-nvidia "$model" "$@"
    return $?
  done
}
